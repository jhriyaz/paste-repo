
apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: backend
  name: backend
spec:
  replicas: {{ .Values.replicaCount }}
  selector:
    matchLabels:
      app: backend
  strategy: {}
  template:
    metadata:
      labels:
        app: backend
    spec:
      containers:
      - image: "{{ .Values.image.backend.repository}}:{{ .Values.image.backend.tag }}"
        name: n-tier-backend
        readinessProbe:
          httpGet:
            path: /
            port: 5000
          initialDelaySeconds: 10
        livenessProbe:
          httpGet:
            path: /
            port: 5000
          initialDelaySeconds: 20
          periodSeconds: 10
        resources:
          requests:
            memory: "50Mi"
            cpu: "100m"
          limits:
            memory: "1Gi"
            cpu: "1"
        envFrom:
          - configMapRef:
              name: mysql-config
